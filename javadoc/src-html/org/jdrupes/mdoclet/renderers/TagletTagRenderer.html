019package org.jdrupes.mdoclet.renderers;
020
021import com.sun.javadoc.Doc;
022import com.sun.javadoc.Tag;
023import com.sun.tools.doclets.Taglet;
024
025import java.util.ArrayList;
026import java.util.HashSet;
027import java.util.List;
028import java.util.Set;
029
030import org.jdrupes.mdoclet.MDoclet;
031
032
033/**
034 * Renders a tag by invocing the taglet's `toString` method.
035 */
036public class TagletTagRenderer implements TagRenderer<Tag> {
037
038        private Taglet taglet;
039        private Set<Doc> handledDocs = new HashSet<>();
040
041        /**
042         * Create a new renderer for the given taglet.
043         * 
044         * @param taglet the taglet
045         */
046    public TagletTagRenderer(Taglet taglet) {
047                this.taglet = taglet;
048        }
049
050        @Override
051    public void render(Tag tag, StringBuilder target, MDoclet doclet) {
052                Doc doc = tag.holder();
053                if (handledDocs.contains(doc)) {
054                        return;
055                }
056                handledDocs.add(doc);
057                List<Tag> tags = new ArrayList<>();
058                for (Tag t: doc.tags()) {
059                        if (t.name().equals(tag.name())) {
060                                tags.add(t);
061                        }
062                }
063        target.append(taglet.toString(tags.toArray(new Tag[tags.size()])));
064    }
065
066}
