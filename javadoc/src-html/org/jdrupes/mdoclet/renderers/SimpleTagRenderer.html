020package org.jdrupes.mdoclet.renderers;
021
022import com.sun.javadoc.Tag;
023
024import org.jdrupes.mdoclet.MDoclet;
025
026import static org.jdrupes.mdoclet.renderers.TagRendering.*;
027
028
029/**
030 * Renders a tag by processing the {@link com.sun.javadoc.Tag#text() text}.
031 */
032public class SimpleTagRenderer implements TagRenderer<Tag> {
033
034    public static final SimpleTagRenderer INSTANCE = new SimpleTagRenderer();
035
036    @Override
037    public void render(Tag tag, StringBuilder target, MDoclet doclet) {
038        target.append(tag.name()).append(" ").append(
039                        simplifySingleParagraph(doclet.toHtml(tag.text())));
040    }
041
042}
