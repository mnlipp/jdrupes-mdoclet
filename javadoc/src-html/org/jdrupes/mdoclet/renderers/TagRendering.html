020package org.jdrupes.mdoclet.renderers;
021
022/**
023 * Utilities for tag rendering.
024 */
025public final class TagRendering {
026
027    private TagRendering() {
028    }
029
030    /**
031     * Removes the `<p>` tag if the given HTML contains only one paragraph.
032     *
033     * **Note:** This implementation may be a bit simplistic: If the HTML starts with a
034     * `<p>` tag, it will be removed. If it ends with `</p>`, this one will be removed,
035     * too. In 99% of the cases, this *exactly* what we wanted. It some special cases,
036     * the result may be invalid HTML -- it should never break things, however, because
037     * HTML is designed to handle "forgotten" tags gracefully.
038     *
039     * @return The HTML without leading `<p>` or trailing `</p>`.
040     */
041    public static String simplifySingleParagraph(String html) {
042        html = html.trim();
043        String upper = html.toUpperCase();
044        if ( upper.startsWith("<P>") ) {
045            html = html.substring(3);
046        }
047        if ( upper.endsWith("</P>") ) {
048            html = html.substring(0, html.length() - 4);
049        }
050        return html;
051    }
052
053}
