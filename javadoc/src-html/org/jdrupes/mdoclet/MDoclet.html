039import com.sun.source.doctree.DocCommentTree;
040
041import jdk.javadoc.doclet.Doclet;
042import jdk.javadoc.doclet.DocletEnvironment;
043import jdk.javadoc.doclet.Reporter;
044import jdk.javadoc.doclet.StandardDoclet;

046/**
047 * The Doclet implementation, which converts the Markdown from the JavaDoc 
048 * comments and tags to HTML.
049 * 
050 * The doclet works by installing wrappers to intercept the 
051 * {@link StandardDoclet}'s calls to access the {@link DocCommentTree}s 
052 * (see {@link DocCommentTreeWrapper}). At the root of this interception
053 * is a modified doclet environment ({@link MDocletEnvironment}) that 
054 * installs a wrapper around doc trees access.
055 * 
056 * For some strange reason, the `StandardDoclet` does not work
057 * with interface {@link DocletEnvironment} but insists on the instance
058 * being a `DocEnvImpl`. Therefore {@link MDocletEnvironment} has
059 * to extend this class which requires to allow module access with
060 * `--add-exports=jdk.javadoc/jdk.javadoc.internal.tool=ALL-UNNAMED`.
061 */
062public class MDoclet implements Doclet {
063
064    public static final String HIGHLIGHT_JS_HTML
065        = "<script type=\"text/javascript\" charset=\"utf-8\" "
066            + "src=\"" + "{@docRoot}/highlight.pack.js" + "\"></script>\n"
067            + "<script type=\"text/javascript\"><!--\n"
068            + "var cssId = 'highlightCss';\n"
069            + "if (!document.getElementById(cssId))\n"
070            + "{\n"
071            + "    var head  = document.getElementsByTagName('head')[0];\n"
072            + "    var link  = document.createElement('link');\n"
073            + "    link.id   = cssId;\n"
074            + "    link.rel  = 'stylesheet';\n"
075            + "    link.type = 'text/css';\n"
076            + "    link.charset = 'utf-8';\n"
077            + "    link.href = '{@docRoot}/highlight.css';\n"
078            + "    link.media = 'all';\n"
079            + "    head.appendChild(link);\n"
080            + "}"
081            + "hljs.initHighlightingOnLoad();\n"
082            + "//--></script>";
083
084    private StandardDoclet standardDoclet;
085    private Reporter reporter;
086    private JavaFileManager fileManager;
087
088    private String markdownProcessorName = FlexmarkProcessor.class.getName();
089    private MarkdownProcessor processor;
090    private List<String> processorOptions = new ArrayList<>();
091    private Option origHeaderOpt;
092    private String bufferedHeader = "";
093    private Option allowScriptsOpt;
094    private boolean disableHighlight;
095    private boolean disableAutoHighlight;
096    private String highlightStyle = "default";
097
098    public MDoclet() {
099        standardDoclet = new StandardDoclet();
100    }
101
102    @Override
103    public void init(Locale locale, Reporter reporter) {
104        this.reporter = reporter;
105        standardDoclet.init(locale, reporter);
106    }
107
108    @Override
109    public String getName() {
110        return getClass().getSimpleName();
111    }
112
113    @Override
114    public Set<? extends Option> getSupportedOptions() {
115        Set<Option> options = new HashSet<>();
116        for (Option opt : standardDoclet.getSupportedOptions()) {
117            if (opt.getNames().contains("-header")) {
118                origHeaderOpt = opt;
119            } else {
120                options.add(opt);
121            }
122            if (opt.getNames().contains("--allow-script-in-comments")) {
123                allowScriptsOpt = opt;
124            }
125        }
126        options.add(new MDocletOption("markdown-processor", 1) {
127            @Override
128            public boolean process(String option, List<String> arguments) {
129                markdownProcessorName = arguments.get(0);
130                return true;
131            }
132        });
133        options.add(new MDocletOption("disable-highlight", 0) {
134            @Override
135            public boolean process(String option, List<String> arguments) {
136                disableHighlight = true;
137                return true;
138            }
139        });
140        options.add(new MDocletOption("disable-auto-highlight", 0) {
141            @Override
142            public boolean process(String option, List<String> arguments) {
143                disableAutoHighlight = true;
144                return true;
145            }
146        });
147        options.add(new MDocletOption("highlight-style", 1) {
148            @Override
149            public boolean process(String option, List<String> arguments) {
150                highlightStyle = arguments.get(0);
151                return true;
152            }
153        });
154        options.add(new MDocletOption("M", 1) {
155            @Override
156            public boolean process(String option, List<String> arguments) {
157                return processorOptions.add(arguments.get(0));
158            }
159        });
160        options.add(new HeaderOverride());
161        return options;
162    }
163
164    private class HeaderOverride implements Option {
165
166        @Override
167        public int getArgumentCount() {
168            return 1;
169        }
170
171        @Override
172        public String getDescription() {
173            return origHeaderOpt.getDescription();
174        }
175
176        @Override
177        public Kind getKind() {
178            return origHeaderOpt.getKind();
179        }
180
181        @Override
182        public List<String> getNames() {
183            return origHeaderOpt.getNames();
184        }
185
186        @Override
187        public String getParameters() {
188            return origHeaderOpt.getParameters();
189        }
190
191        @Override
192        public boolean process(String option, List<String> arguments) {
193            bufferedHeader = arguments.get(0);
194            return true;
195        }
196
197    }
198
199    @Override
200    public SourceVersion getSupportedSourceVersion() {
201        return SourceVersion.latest();
202    }
203
204    @Override
205    public boolean run(DocletEnvironment environment) {
206        fileManager = environment.getJavaFileManager();
207        if (disableHighlight) {
208            if (bufferedHeader.length() > 0) {
209                origHeaderOpt.process("-header", List.of(bufferedHeader));
210            }
211        } else {
212            bufferedHeader += HIGHLIGHT_JS_HTML;
213            origHeaderOpt.process("-header", List.of(bufferedHeader));
214            allowScriptsOpt.process("--allow-script-in-comments",
215                Collections.emptyList());
216        }
217
218        MDocletEnvironment env = new MDocletEnvironment(this, environment);
219        processor = createProcessor();
220        processor.start(processorOptions.toArray(new String[0]));
221        return standardDoclet.run(env) && postProcess();
222    }
223
224    private MarkdownProcessor createProcessor() {
225        try {
226            @SuppressWarnings("unchecked")
227            Class<MarkdownProcessor> mpc = (Class<MarkdownProcessor>) getClass()
228                .getClassLoader().loadClass(markdownProcessorName);
229            MarkdownProcessor mdp = (MarkdownProcessor) mpc
230                .getDeclaredConstructor().newInstance();
231            if (disableAutoHighlight && mdp.isSupportedOption(
232                MarkdownProcessor.INTERNAL_OPT_DISABLE_AUTO_HIGHLIGHT) >= 0) {
233                processorOptions
234                    .add(MarkdownProcessor.INTERNAL_OPT_DISABLE_AUTO_HIGHLIGHT);
235            }
236            return mdp;
237        } catch (ClassNotFoundException | InstantiationException
238                | IllegalAccessException | ClassCastException
239                | IllegalArgumentException | InvocationTargetException
240                | NoSuchMethodException | SecurityException e) {
241            reporter.print(Diagnostic.Kind.ERROR,
242                "Markdown processor \"" + markdownProcessorName
243                    + "\" cannot be loaded (" + e.getMessage()
244                    + "), check name and docletpath");
245            return null;
246        }
247    }
248
249    /**
250     * Returns the processor selected by the options.
251     * 
252     * @return the processor
253     */
254    public MarkdownProcessor getProcessor() {
255        return processor;
256    }
257
258    private boolean postProcess() {
259        if (disableHighlight) {
260            return true;
261        }
262        return copyResource("highlight.pack.js", "highlight.pack.js",
263            "highlight.js")
264            && copyResource("highlight-LICENSE.txt", "highlight-LICENSE.txt",
265                "highlight.js license")
266            && copyResource("highlight-styles/" + highlightStyle + ".css",
267                "highlight.css",
268                "highlight.js style '" + highlightStyle + "'");
269    }
270
271    private boolean copyResource(String resource, String destination,
272            String description) {
273        try {
274            InputStream in = MDoclet.class.getResourceAsStream(resource);
275            if (in == null) {
276                throw new FileNotFoundException();
277            }
278            in.transferTo(
279                fileManager.getFileForOutput(Location.DOCUMENTATION_OUTPUT, "",
280                    destination, null).openOutputStream());
281            return true;
282        } catch (IOException e) {
283            reporter.print(javax.tools.Diagnostic.Kind.ERROR,
284                "Error writing " + description + ": "
285                    + e.getLocalizedMessage());
286            return false;
287        }
288    }
