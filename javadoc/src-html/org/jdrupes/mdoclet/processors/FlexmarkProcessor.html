018package org.jdrupes.mdoclet.processors;

020import java.lang.reflect.InvocationTargetException;
021import java.util.ArrayList;
022import java.util.HashSet;
023import java.util.List;
024import java.util.Set;
025
026import org.jdrupes.mdoclet.MarkdownProcessor;
027import org.jdrupes.mdoclet.processors.flexmark.TopAnchorLinkExtension;
028
029import com.vladsch.flexmark.Extension;
030import com.vladsch.flexmark.ast.Node;
031import com.vladsch.flexmark.ext.abbreviation.AbbreviationExtension;
032import com.vladsch.flexmark.ext.anchorlink.AnchorLinkExtension;
033import com.vladsch.flexmark.ext.definition.DefinitionExtension;
034import com.vladsch.flexmark.ext.footnotes.FootnoteExtension;
035import com.vladsch.flexmark.ext.tables.TablesExtension;
036import com.vladsch.flexmark.ext.toc.TocExtension;
037import com.vladsch.flexmark.ext.typographic.TypographicExtension;
038import com.vladsch.flexmark.ext.wikilink.WikiLinkExtension;
039import com.vladsch.flexmark.html.HtmlRenderer;

042import com.vladsch.flexmark.util.options.MutableDataSet;

044/**
045 * This class provides an adapter for the 
046 * [flexmark-java](https://github.com/vsch/flexmark-java) Markdown
047 * processor.
048 * 
049 * The adapter supports the following flags:
050 * 
051 * `-parser-profile` 
052 * :   Sets one of the profiles defined in
053 *     `com.vladsch.flexmark.parser.ParserEmulationProfile`. The name of the
054 *     profle is the lower case name of the enum value. At the time of this
055 *     writing supported names are:
056 *      - commonmark (default)
057 *      - fixed_indent
058 *      - kramdown
059 *      - markdown
060 *      - github_doc
061 *      - multi_markdown
062 *      - pegdown
063 * 
064 * `-clear-extensions`
065 * :   Clears the list of extensions. The following extensions are predefined:
066 *       - [Abbreviation](https://github.com/vsch/flexmark-java/wiki/Extensions#abbreviation)
067 *       - [AnchorLink](https://github.com/vsch/flexmark-java/wiki/Extensions#anchorlink)
068 *       - [Definition](https://github.com/vsch/flexmark-java/wiki/Extensions#definition-lists) 
069 *         (Definition Lists)[^DefLists]
070 *       - [Footnote](https://github.com/vsch/flexmark-java/wiki/Extensions#footnotes)
071 *       - [Tables](https://github.com/vsch/flexmark-java/wiki/Extensions#tables)
072 *       - [Table of Content](https://github.com/vsch/flexmark-java/wiki/Extensions#table-of-contents-1)
073 *       - {@link TopAnchorLinkExtension TopAnchorLink} (provided by MDoclet)
074 * 
075 * `-extension <name>`
076 * :   Adds the flexmark extension with the given name to the list of extensions.
077 *     If the name contains a dot, it is assumed to be a fully qualified class
078 *     name. Else, it is expanded using the naming pattern used by flexmark.
079 *     
080 * The parser also supports disabling the automatic highlight feature.
081 * 
082 * [^DefLists]: If you use this extension, you'll most likely want to supply a
083 *     modified style sheet because the standard stylesheet assumes all definition
084 *     lists to be parameter defintion lists and formats them accordingly.
085 *     
086 *     Here are the changes made for this documentation:
087 *     ```css
088 *     /* [MOD] {@literal *}/
089 *     /* .contentContainer .description dl dd, {@literal *}/ .contentContainer .details dl dt, .serializedFormContainer dl dt {
090 *         font-size:12px;
091 *         font-weight:bold;
092 *         margin:10px 0 0 0;
093 *         color:#4E4E4E;
094 *     }
095 *     /* [MOD] Added {@literal *}/
096 *     dl dt {
097 *         margin:10px 0 0 0;
098 *     }
099 *     
100 *     /* [MOD] {@literal *}/
101 *     /* .contentContainer .description dl dd, {@literal *}/ .contentContainer .details dl dd, .serializedFormContainer dl dd {
102 *         margin:5px 0 10px 0px;
103 *         font-size:14px;
104 *         font-family:'DejaVu Sans Mono',monospace;
105 *     }
106 *     ```
107 * 
108 */
109public class FlexmarkProcessor implements MarkdownProcessor {
110
111        final private static String OPT_PROFILE = "-parser-profile";
112        final private static String OPT_CLEAR_EXTENSIONS = "-clear-extensions";
113        final private static String OPT_EXTENSION = "-extension";
114        
115        private Parser parser;
116        private HtmlRenderer renderer;
117        
118        @Override
119        public int isSupportedOption(String option) {
120                switch (option) {
121                case OPT_CLEAR_EXTENSIONS:
122                case INTERNAL_OPT_DISABLE_AUTO_HIGHLIGHT:
123                        return 0;
124                        
125                case OPT_PROFILE:
126                case OPT_EXTENSION:
127                        return 1;
128                default:
129                        return -1;
130                }
131        }
132
133
134        @Override
135        public void start(String[][] options) {
136        MutableDataSet flexmarkOpts = new MutableDataSet();
137        Set<Class<? extends Extension>> extensions = new HashSet<>();
138        extensions.add(AbbreviationExtension.class);
139        extensions.add(AnchorLinkExtension.class);
140        extensions.add(DefinitionExtension.class);
141        extensions.add(FootnoteExtension.class);
142        extensions.add(TablesExtension.class);
143        extensions.add(TypographicExtension.class);
144        extensions.add(TocExtension.class);
145        extensions.add(WikiLinkExtension.class);
146        extensions.add(TopAnchorLinkExtension.class);
147        
148        flexmarkOpts.set(HtmlRenderer.GENERATE_HEADER_ID, true);
149        
150        for (String[] opt: options) {
151                switch (opt[0]) {
152                case OPT_PROFILE:
153                        setFromProfile(flexmarkOpts, opt[1]);
154                        continue;
155                        
156                case OPT_CLEAR_EXTENSIONS:
157                        extensions.clear();
158                        continue;
159                        
160                case OPT_EXTENSION:
161                        try {
162                                String clsName = opt[1];
163                                if (!clsName.contains(".")) {
164                                        clsName = "com.vladsch.flexmark.ext."
165                                                + opt[1].toLowerCase() + "." + opt[1] + "Extension";
166                                }
167                                @SuppressWarnings("unchecked") 
168                                Class<? extends Extension> cls = (Class<? extends Extension>) 
169                                        getClass().getClassLoader().loadClass(clsName);
170                                extensions.add(cls);
171                                continue;
172                        } catch (ClassNotFoundException | ClassCastException e) {
173                                throw new IllegalArgumentException
174                                        ("Cannot find extension " + opt[1]
175                                                + " (check spelling and classpath).");
176                        }
177                
178                case INTERNAL_OPT_DISABLE_AUTO_HIGHLIGHT:
179                        flexmarkOpts.set
180                                (HtmlRenderer.FENCED_CODE_NO_LANGUAGE_CLASS, "nohighlight");
181                        continue;
182
183                default:
184                throw new IllegalArgumentException("Unknown option: " + opt[0]);
185                }
186        }
187        
188        List<Extension> extObjs = new ArrayList<>();
189        for (Class<? extends Extension> cls: extensions) {
190                try {
191                                extObjs.add((Extension)cls.getMethod("create").invoke(null));
192                        } catch (IllegalAccessException | IllegalArgumentException 
193                                        | InvocationTargetException | NoSuchMethodException 
194                                        | SecurityException e) {
195                                throw new IllegalArgumentException("Cannot create extension of type "
196                                                + cls + ".");
197                        }
198        }
199        if (!extObjs.isEmpty()) {
200                flexmarkOpts.set(Parser.EXTENSIONS, extObjs);
201        }
202        parser = Parser.builder(flexmarkOpts).build();
203        renderer = HtmlRenderer.builder(flexmarkOpts).build();
204        }
205
206        private void setFromProfile(MutableDataSet fmOpts, String profileName) {
207                for (ParserEmulationProfile p: ParserEmulationProfile.values()) {
208                        if (p.toString().equalsIgnoreCase(profileName)) {
209                                fmOpts.setFrom(p);
210                                return;
211                        }
212                }
213                throw new IllegalArgumentException("Unknown profile: " + profileName);
214        }
215
216
217        /* (non-Javadoc)
218         * @see org.jdrupes.mdoclet.MarkdownProcessor#toHtml(java.lang.String)
219         */
220        @Override
221        public String toHtml(String markdown) {
222        Node document = parser.parse(markdown);
223        return renderer.render(document);
224        }
225
