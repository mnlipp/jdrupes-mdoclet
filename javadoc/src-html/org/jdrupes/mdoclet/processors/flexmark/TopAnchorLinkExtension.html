018package org.jdrupes.mdoclet.processors.flexmark;
019
020import com.vladsch.flexmark.Extension;
021import com.vladsch.flexmark.ast.Link;
022import com.vladsch.flexmark.ast.Node;
023import com.vladsch.flexmark.html.AttributeProvider;
024import com.vladsch.flexmark.html.HtmlRenderer;
025import com.vladsch.flexmark.html.IndependentAttributeProviderFactory;
026import com.vladsch.flexmark.html.HtmlRenderer.Builder;
027import com.vladsch.flexmark.html.renderer.AttributablePart;
028import com.vladsch.flexmark.html.renderer.NodeRendererContext;
029import com.vladsch.flexmark.util.html.Attributes;
030import com.vladsch.flexmark.util.options.MutableDataHolder;
031
032/**
033 * Provides an extension that adds "`target='_top'`" to all anchor links.
034 */
035public class TopAnchorLinkExtension implements HtmlRenderer.HtmlRendererExtension {
036
037        /**
038         * Called with all options. Currently this extension supports no options and
039         * does nothing.
040         */
041        @Override
042        public void rendererOptions(MutableDataHolder options) {
043                // Currently no options supported.
044        }
045
046        /**
047         * This method is called once when the builder for the output is created.
048         * It must modify the passed in builder as required by the extension.  
049         */
050        @Override
051        public void extend(Builder rendererBuilder, String rendererType) {
052                rendererBuilder.attributeProviderFactory(new IndependentAttributeProviderFactory() {
053                        
054                        @Override
055                        public AttributeProvider create(NodeRendererContext context) {
056                                return new AttributeProvider() {
057
058                                        @Override
059                                        public void setAttributes(Node node, AttributablePart part, 
060                                                        Attributes attributes) {
061                                                if (node instanceof Link && part == AttributablePart.LINK) {
062                                            attributes.replaceValue("target", "_top");
063                                        }
064                                        }
065                                        
066                                };
067                        }
068                });
069        }
070        
071        /**
072         * Extensions are added by providing a class to flexmark. Flexmark then
073         * invokes the extension's static `create` method through reflection.
074         * 
075         * @return an instance of the extension
076         */
077        public static Extension create() {
078                return new TopAnchorLinkExtension();
079        }
080}
