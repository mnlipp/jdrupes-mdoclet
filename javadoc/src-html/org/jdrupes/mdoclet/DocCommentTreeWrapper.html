019package org.jdrupes.mdoclet;
020
021import java.util.ArrayList;
022import java.util.List;
023
024import com.sun.source.doctree.DocCommentTree;
025import com.sun.source.doctree.DocTree;
026import com.sun.source.doctree.DocTreeVisitor;
027import com.sun.tools.javac.tree.DCTree;
028
029/**
030 * Feeds the results of the methods through the {@link TreeConverter}
031 * where appropriate.
032 * 
033 * With JDK 11, {@link DocCommentTreeWrapper} only had to implement
034 * {@link DocCommentTree}. Starting with (at least) JDK 15, javadoc
035 * casts (without check) {@link DocCommentTree} instances to
036 * {@link DCTree.DCDocComment} when reporting errors e.g. about
037 * a missing link target. That's the only reason why the
038 * wrapper extends {@link DCTree.DCDocComment}.
039 */
040public class DocCommentTreeWrapper extends DCTree.DCDocComment 
041    implements DocCommentTree {
042
043    private DocCommentTree tree;
044    private TreeConverter treeConverter;
045
046    public DocCommentTreeWrapper(MDoclet doclet, MDocletEnvironment environment,
047            DocCommentTree tree) {
048        super(tree instanceof DCTree.DCDocComment 
049            ? ((DCTree.DCDocComment)tree).comment : null, 
050            null, null, null, null, null, null);
051        this.tree = tree;
052        treeConverter = new TreeConverter(doclet.getProcessor(),
053            environment.getDocTrees().getDocTreeFactory());
054    }
055
056    /**
057     * Overridden by {@link DCTree.DCDocComment}, restore to default.
058     */
059    @Override
060    public List<? extends DocTree> getFullBody() {
061        ArrayList<DocTree> bodyList = new ArrayList<>();
062        bodyList.addAll(getFirstSentence());
063        bodyList.addAll(getBody());
064        return bodyList;
065    }
066    
067    /**
068     * {@inheritDoc}
069     * 
070     * @see com.sun.source.doctree.DocCommentTree#getFirstSentence()
071     */
072    public List<? extends DocTree> getFirstSentence() {
073        return treeConverter.convertFragment(tree.getFirstSentence());
074    }
075
076    /**
077     * {@inheritDoc}
078     * 
079     * @see com.sun.source.doctree.DocCommentTree#getBody()
080     */
081    public List<? extends DocTree> getBody() {
082        return treeConverter.convertDescription(tree.getBody());
083    }
084
085    /**
086     * {@inheritDoc}
087     * 
088     * @see com.sun.source.doctree.DocCommentTree#getBlockTags()
089     */
090    public List<? extends DocTree> getBlockTags() {
091        List<? extends DocTree> blockTags = tree.getBlockTags();
092        List<DocTree> result = new ArrayList<>();
093        for (DocTree tree : blockTags) {
094            treeConverter.convertTag(result, tree);
095        }
096        return result;
097    }
098
099    /**
100     * {@inheritDoc}
101     * 
102     * @see com.sun.source.doctree.DocCommentTree#getPreamble()
103     */
104    public List<? extends DocTree> getPreamble() {
105        return tree.getPreamble();
106    }
107
108    /**
109     * {@inheritDoc}
110     * 
111     * @see com.sun.source.doctree.DocCommentTree#getPostamble()
112     */
113    public List<? extends DocTree> getPostamble() {
114        return tree.getPostamble();
115    }
116
117    /**
118     * {@inheritDoc}
119     * 
120     * @see com.sun.source.doctree.DocTree#getKind()
121     */
122    public Kind getKind() {
123        return tree.getKind();
124    }
125
126    /**
127     * {@inheritDoc}
128     * 
129     * @see com.sun.source.doctree.DocTree#accept(com.sun.source.doctree.DocTreeVisitor, java.lang.Object)
130     */
131    public <R, D> R accept(DocTreeVisitor<R, D> visitor, D data) {
132        return tree.accept(visitor, data);
133    }
134
135}
