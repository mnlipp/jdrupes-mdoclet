019package org.jdrupes.mdoclet.processors.flexmark;
020
021import com.vladsch.flexmark.Extension;
022import com.vladsch.flexmark.ast.Link;
023import com.vladsch.flexmark.ast.Node;
024import com.vladsch.flexmark.html.AttributeProvider;
025import com.vladsch.flexmark.html.HtmlRenderer;
026import com.vladsch.flexmark.html.HtmlRenderer.Builder;
027import com.vladsch.flexmark.html.IndependentAttributeProviderFactory;
028import com.vladsch.flexmark.html.renderer.AttributablePart;
029import com.vladsch.flexmark.html.renderer.NodeRendererContext;
030import com.vladsch.flexmark.util.html.Attributes;
031import com.vladsch.flexmark.util.options.MutableDataHolder;
032
033/**
034 * Provides an extension that adds "`target='_top'`" to all anchor links.
035 */
036public class TopAnchorLinkExtension
037        implements HtmlRenderer.HtmlRendererExtension {
038
039    /**
040     * Called with all options. Currently this extension supports no options and
041     * does nothing.
042     */
043    @Override
044    public void rendererOptions(MutableDataHolder options) {
045        // Currently no options supported.
046    }
047
048    /**
049     * This method is called once when the builder for the output is created.
050     * It must modify the passed in builder as required by the extension.  
051     */
052    @Override
053    public void extend(Builder rendererBuilder, String rendererType) {
054        rendererBuilder.attributeProviderFactory(
055            new IndependentAttributeProviderFactory() {
056
057                @Override
058                public AttributeProvider create(NodeRendererContext context) {
059                    return new AttributeProvider() {
060
061                        @Override
062                        public void setAttributes(Node node,
063                                AttributablePart part,
064                                Attributes attributes) {
065                            if (node instanceof Link
066                                && part == AttributablePart.LINK) {
067                                attributes.replaceValue("target", "_top");
068                            }
069                        }
070
071                    };
072                }
073            });
074    }
075
076    /**
077     * Extensions are added by providing a class to flexmark. Flexmark then
078     * invokes the extension's static `create` method through reflection.
079     * 
080     * @return an instance of the extension
081     */
082    public static Extension create() {
083        return new TopAnchorLinkExtension();
084    }
085}
