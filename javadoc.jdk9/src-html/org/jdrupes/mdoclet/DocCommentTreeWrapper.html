019package org.jdrupes.mdoclet;
020
021import java.util.ArrayList;
022import java.util.List;
023
024import com.sun.source.doctree.DocCommentTree;
025import com.sun.source.doctree.DocTree;
026import com.sun.source.doctree.DocTreeVisitor;
027
028/**
029 * Feeds the results of the methods through the {@link TreeConverter}
030 * where appropriate.
031 */
032public class DocCommentTreeWrapper implements DocCommentTree {
033
034    private DocCommentTree tree;
035    private TreeConverter treeConverter;
036
037    public DocCommentTreeWrapper(MDoclet doclet, MDocletEnvironment environment,
038            DocCommentTree tree) {
039        super();
040        this.tree = tree;
041        treeConverter = new TreeConverter(doclet.getProcessor(),
042            environment.getDocTrees().getDocTreeFactory());
043    }
044
045    /**
046     * {@inheritDoc}
047     * 
048     * @see com.sun.source.doctree.DocCommentTree#getFirstSentence()
049     */
050    public List<? extends DocTree> getFirstSentence() {
051        return treeConverter.convertFragment(tree.getFirstSentence());
052    }
053
054    /**
055     * {@inheritDoc}
056     * 
057     * @see com.sun.source.doctree.DocCommentTree#getBody()
058     */
059    public List<? extends DocTree> getBody() {
060        return treeConverter.convertDescription(tree.getBody());
061    }
062
063    /**
064     * {@inheritDoc}
065     * 
066     * @see com.sun.source.doctree.DocCommentTree#getBlockTags()
067     */
068    public List<? extends DocTree> getBlockTags() {
069        List<? extends DocTree> blockTags = tree.getBlockTags();
070        List<DocTree> result = new ArrayList<>();
071        for (DocTree tree : blockTags) {
072            treeConverter.convertTag(result, tree);
073        }
074        return result;
075    }
076
077    /**
078     * {@inheritDoc}
079     * 
080     * @see com.sun.source.doctree.DocCommentTree#getPreamble()
081     */
082    public List<? extends DocTree> getPreamble() {
083        return tree.getPreamble();
084    }
085
086    /**
087     * {@inheritDoc}
088     * 
089     * @see com.sun.source.doctree.DocCommentTree#getPostamble()
090     */
091    public List<? extends DocTree> getPostamble() {
092        return tree.getPostamble();
093    }
094
095    /**
096     * {@inheritDoc}
097     * 
098     * @see com.sun.source.doctree.DocTree#getKind()
099     */
100    public Kind getKind() {
101        return tree.getKind();
102    }
103
104    /**
105     * {@inheritDoc}
106     * 
107     * @see com.sun.source.doctree.DocTree#accept(com.sun.source.doctree.DocTreeVisitor, java.lang.Object)
108     */
109    public <R, D> R accept(DocTreeVisitor<R, D> visitor, D data) {
110        return tree.accept(visitor, data);
111    }
112
113}
