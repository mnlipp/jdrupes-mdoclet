052/**
053 * Wraps the {@link DocTrees} passed to the constructor.
054 * 
055 * The main purpose of the wrapper is to wrap the results from methods
056 * that return a {@link DocCommentTree} for already parsed content
057 * in a {@link DocCommentTreeWrapper}.
058 * 
059 * For method invocations with {@link FileObject}s as argument,
060 * the source is converted from markdown to html if the name ends
061 * with ".md".
062 */
063public class DocTreesWrapper extends DocTrees {
064
065    private final MDoclet doclet;
066    private final MDocletEnvironment environment;
067    private final DocTrees docTrees;
068
069    public DocTreesWrapper(MDoclet doclet, MDocletEnvironment environment,
070            DocTrees docTrees) {
071        super();
072        this.doclet = doclet;
073        this.environment = environment;
074        this.docTrees = docTrees;
075    }
076
077    private DocCommentTree wrap(DocCommentTree tree) {
078        if (tree == null) {
079            return null;
080        }
081        return new DocCommentTreeWrapper(doclet, environment, tree);
082    }

125    private FileObject wrapMdFile(FileObject fileObject) {
126        if (!fileObject.getName().endsWith(".md")) {
127            return fileObject;
128        }
129        return new ForwardingFileObject<>(fileObject) {
130
131            @Override
132            public String getName() {
133                String origName = super.getName();
134                return origName.substring(0, origName.length() - 2)
135                    + "html";
136            }
137
138            @Override
139            public CharSequence getCharContent(boolean ignoreEncodingErrors)
140                    throws IOException {
141                String md
142                    = super.getCharContent(ignoreEncodingErrors).toString();
143                String html = doclet.getProcessor().toHtml(md);
144                return "<body>" + html + "</body>";
145            }
146        };
147    }
