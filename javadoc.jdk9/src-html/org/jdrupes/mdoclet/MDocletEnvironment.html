019package org.jdrupes.mdoclet;
020
021import java.util.Set;
022
023import javax.lang.model.SourceVersion;
024import javax.lang.model.element.Element;
025import javax.lang.model.element.TypeElement;
026import javax.lang.model.util.Elements;
027import javax.lang.model.util.Types;
028import javax.tools.JavaFileManager;
029import javax.tools.JavaFileObject.Kind;
030
031import com.sun.source.util.DocTrees;
032
033import jdk.javadoc.doclet.DocletEnvironment;
034import jdk.javadoc.internal.tool.DocEnvImpl;
035
036/**
037 * Wraps the {@link DocEnvImpl} passed to the doclet for the
038 * sole purpose of inserting a wrapper for the doctrees access.
039 */
040public class MDocletEnvironment extends DocEnvImpl
041        implements DocletEnvironment {
042
043    DocletEnvironment defaultEnvironment;
044    DocTreesWrapper docTrees;
045
046    public MDocletEnvironment(MDoclet doclet, DocletEnvironment environment) {
047        super(((DocEnvImpl) environment).toolEnv,
048            ((DocEnvImpl) environment).etable);
049        this.defaultEnvironment = environment;
050        docTrees = new DocTreesWrapper(doclet, this, environment.getDocTrees());
051    }
052
053    /**
054     * Delegates to the environment passed in the constructor.
055     * 
056     * {@inheritDoc}
057     * @see jdk.javadoc.doclet.DocletEnvironment#getSpecifiedElements()
058     */
059    public Set<? extends Element> getSpecifiedElements() {
060        return defaultEnvironment.getSpecifiedElements();
061    }
062
063    /**
064     * Delegates to the environment passed in the constructor.
065     * 
066     * {@inheritDoc}
067     * @see jdk.javadoc.doclet.DocletEnvironment#getIncludedElements()
068     */
069    public Set<? extends Element> getIncludedElements() {
070        return defaultEnvironment.getIncludedElements();
071    }
072
073    /**
074     * Wraps the doctrees from the environment passed in the constructor
075     * in a {@link DocTreesWrapper}.
076     * 
077     * {@inheritDoc}
078     * @see jdk.javadoc.doclet.DocletEnvironment#getDocTrees()
079     */
080    public DocTrees getDocTrees() {
081        return docTrees;
082    }
083
084    /**
085     * Delegates to the environment passed in the constructor.
086     * 
087     * {@inheritDoc}
088     * @see jdk.javadoc.doclet.DocletEnvironment#getElementUtils()
089     */
090    public Elements getElementUtils() {
091        return defaultEnvironment.getElementUtils();
092    }
093
094    /**
095     * Delegates to the environment passed in the constructor.
096     * 
097     * {@inheritDoc}
098     * @see jdk.javadoc.doclet.DocletEnvironment#getTypeUtils()
099     */
100    public Types getTypeUtils() {
101        return defaultEnvironment.getTypeUtils();
102    }
103
104    /**
105     * Delegates to the environment passed in the constructor.
106     * 
107     * {@inheritDoc}
108     * @see jdk.javadoc.doclet.DocletEnvironment#isIncluded(javax.lang.model.element.Element)
109     */
110    public boolean isIncluded(Element e) {
111        return defaultEnvironment.isIncluded(e);
112    }
113
114    /**
115     * Delegates to the environment passed in the constructor.
116     * 
117     * {@inheritDoc}
118     * @see jdk.javadoc.doclet.DocletEnvironment#isSelected(javax.lang.model.element.Element)
119     */
120    public boolean isSelected(Element e) {
121        return defaultEnvironment.isSelected(e);
122    }
123
124    /**
125     * Delegates to the environment passed in the constructor.
126     * 
127     * {@inheritDoc}
128     * @see jdk.javadoc.doclet.DocletEnvironment#getJavaFileManager()
129     */
130    public JavaFileManager getJavaFileManager() {
131        return defaultEnvironment.getJavaFileManager();
132    }
133
134    /**
135     * Delegates to the environment passed in the constructor.
136     * 
137     * {@inheritDoc}
138     * @see jdk.javadoc.doclet.DocletEnvironment#getSourceVersion()
139     */
140    public SourceVersion getSourceVersion() {
141        return defaultEnvironment.getSourceVersion();
142    }
143
144    /**
145     * Delegates to the environment passed in the constructor.
146     * 
147     * @return the module mode
148     * @see jdk.javadoc.doclet.DocletEnvironment#getModuleMode()
149     */
150    public ModuleMode getModuleMode() {
151        return defaultEnvironment.getModuleMode();
152    }
153
154    /**
155     * Delegates to the environment passed in the constructor.
156     * 
157     * {@inheritDoc}
158     * @see jdk.javadoc.doclet.DocletEnvironment#getFileKind(javax.lang.model.element.TypeElement)
159     */
160    public Kind getFileKind(TypeElement type) {
161        return defaultEnvironment.getFileKind(type);
162    }
163
164}
