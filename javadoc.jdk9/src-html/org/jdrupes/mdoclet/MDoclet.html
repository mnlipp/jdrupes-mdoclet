019package org.jdrupes.mdoclet;
020
021import java.lang.reflect.InvocationTargetException;
022import java.util.ArrayList;
023import java.util.HashSet;
024import java.util.List;
025import java.util.Locale;
026import java.util.Set;
027
028import javax.lang.model.SourceVersion;
029import javax.tools.Diagnostic;
030
031import org.jdrupes.mdoclet.processors.FlexmarkProcessor;
032
033import com.sun.source.doctree.DocCommentTree;
034
035import jdk.javadoc.doclet.Doclet;
036import jdk.javadoc.doclet.DocletEnvironment;
037import jdk.javadoc.doclet.Reporter;
038import jdk.javadoc.doclet.StandardDoclet;
039
040/**
041 * The Doclet implementation, which converts the Markdown from the JavaDoc 
042 * comments and tags to HTML.
043 * 
044 * The doclet works by installing wrappers to intercept the 
045 * {@link StandardDoclet}'s calls to access the {@link DocCommentTree}s 
046 * (see {@link DocCommentTreeWrapper}). At the root of this interception
047 * is a modified doclet environment ({@link MDocletEnvironment}) that 
048 * installs a wrapper around doc trees access.
049 * 
050 * For some strange reason, the `StandardDoclet` does not work
051 * with interface {@link DocletEnvironment} but insists on the instance
052 * being a `DocEnvImpl`. Therefore {@link MDocletEnvironment} has
053 * to extend this class which requires to allow module access with
054 * `--add-exports=jdk.javadoc/jdk.javadoc.internal.tool=ALL-UNNAMED`.
055 * 
056 * @version 2.0
057 * @see <a href='https://openjdk.java.net/groups/compiler/using-new-doclet.html'>Using the new doclet API</a>
058 */
059public class MDoclet implements Doclet {
060
061    private StandardDoclet standardDoclet;
062    private Reporter reporter;
063
064    private String markdownProcessorName = FlexmarkProcessor.class.getName();
065    private MarkdownProcessor processor;
066    private List<String> processorOptions = new ArrayList<>();
067
068    public MDoclet() {
069        standardDoclet = new StandardDoclet();
070    }
071
072    @Override
073    public void init(Locale locale, Reporter reporter) {
074        this.reporter = reporter;
075        standardDoclet.init(locale, reporter);
076    }
077
078    @Override
079    public String getName() {
080        return getClass().getSimpleName();
081    }
082
083    @Override
084    public Set<? extends Option> getSupportedOptions() {
085        Set<Option> options
086            = new HashSet<>(standardDoclet.getSupportedOptions());
087        options.add(new MDocletOption("markdown-processor", 1) {
088            @Override
089            public boolean process(String option, List<String> arguments) {
090                markdownProcessorName = arguments.get(0);
091                return true;
092            }
093        });
094        options.add(new MDocletOption("disable-highlight", 0) {
095            @Override
096            public boolean process(String option, List<String> arguments) {
097                return true;
098            }
099        });
100        options.add(new MDocletOption("disable-auto-highlight", 0) {
101            @Override
102            public boolean process(String option, List<String> arguments) {
103                return true;
104            }
105        });
106        options.add(new MDocletOption("highlight-style", 1) {
107            @Override
108            public boolean process(String option, List<String> arguments) {
109                return true;
110            }
111        });
112        options.add(new MDocletOption("M", 1) {
113            @Override
114            public boolean process(String option, List<String> arguments) {
115                return processorOptions.add(arguments.get(0));
116            }
117        });
118        return options;
119    }
120
121    @Override
122    public SourceVersion getSupportedSourceVersion() {
123        return SourceVersion.latest();
124    }
125
126    @Override
127    public boolean run(DocletEnvironment environment) {
128        MDocletEnvironment env = new MDocletEnvironment(this, environment);
129        processor = createProcessor();
130        processor.start(processorOptions.toArray(new String[0]));
131        boolean result = standardDoclet.run(env);
132        return result;
133    }
134
135    private MarkdownProcessor createProcessor() {
136        try {
137            @SuppressWarnings("unchecked")
138            Class<MarkdownProcessor> mpc = (Class<MarkdownProcessor>) getClass()
139                .getClassLoader().loadClass(markdownProcessorName);
140            return (MarkdownProcessor) mpc.getDeclaredConstructor()
141                .newInstance();
142        } catch (ClassNotFoundException | InstantiationException
143                | IllegalAccessException | ClassCastException
144                | IllegalArgumentException | InvocationTargetException
145                | NoSuchMethodException | SecurityException e) {
146            reporter.print(Diagnostic.Kind.ERROR,
147                "Markdown processor \"" + markdownProcessorName
148                    + "\" cannot be loaded (" + e.getMessage()
149                    + "), check name and docletpath");
150            return null;
151        }
152    }
153
154    /**
155     * Returns the processor selected by the options.
156     * 
157     * @return the processor
158     */
159    public MarkdownProcessor getProcessor() {
160        return processor;
161    }
162
163}
